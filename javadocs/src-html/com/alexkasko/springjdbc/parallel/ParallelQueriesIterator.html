030    /**
031     * Executes single SQL query to multiple data sources in parallel using provided executor.
032     * Provides data to client as iterator, {@link ArrayBlockingQueue} is used for temporary buffering.
033     * Iteration will block awaiting data loaded from sources.
034     * Typical usage is to get new instance somewhere (spring prototype bean etc.), provide query params
035     * with <code>start</code> method and iterate over until end.
036     * Data source exceptions will be propagated propagated to caller (iterating) thread and given to {@link ParallelQueriesExceptionHandler}.
037     * All parallel queries will be cancelled on one query error, propagated by exception handler.
038     * Iterator takes sql in the same format as {@code NamedParameterJdbcTemplate} does (with {@code :palceholders})
039     * and map declared parameters to provided values using the same methods as {@code NamedParameterJdbcTemplate}.
040     * <b>NOT</b> thread-safe (tbd: specify points that break thread safety), instance may be reused calling <code>start</code> method, but only in one thread simultaneously.
041     *
042     * @author  alexkasko
043     * Date: 6/8/12
044     * @see ParallelQueriesListener
045     * @see com.alexkasko.springjdbc.parallel.accessor.DataSourceAccessor
046     */
047    
048    public class ParallelQueriesIterator<T> extends AbstractIterator<T> {
049        private static final Log logger = LogFactory.getLog(ParallelQueriesIterator.class);
050    
051        private final Object endOfDataObject = new Object();
052        private final ExceptionHolder exceptionHolder = new ExceptionHolder();
053    
054        private final DataSourceAccessor<?, ?> sources;
055        private final String sql;
056        private final RowMapperFactory<T, ?> mapperFactory;
057        private final ExecutorService executor;
058        // was made non-generic to allow endOfDataObject
059        private final ArrayBlockingQueue<Object> dataQueue;
060        private final List<ParallelQueriesListener> listeners = Lists.newArrayList();
061        private final int maxDataPollWaitSeconds;
062        private final ParallelQueriesExceptionHandler exceptionHandler;
063    
064        private AtomicBoolean started = new AtomicBoolean(false);
065        private AtomicInteger sourcesRemained = new AtomicInteger(0);
066        // this field needed anyway in start method, so we'll use it in workers instead of thread.isInterrupted
067        private final AtomicBoolean cancelled = new AtomicBoolean(false);
068        private final ConcurrentHashMap<String, Statement> activeStatements = new ConcurrentHashMap<String, Statement>();
069        private ImmutableList<Future<?>> futures;
070    
071        /**
072         * Shortcut constructor, uses {@link RoundRobinAccessor}, new cached thread pool,
073         * and {@code JdbcTemplate}s with default settings
074         *
075         *
076         * @param sources list of data sources, will be used in round-robin mode
077         * @param sql query to execute using JdbcTemplate
078         * @param mapper will be used to get data from result sets
079         */
080        public ParallelQueriesIterator(Collection<DataSource> sources, String sql, RowMapper<T> mapper) {
081            this(RoundRobinAccessor.of(Collections2.transform(sources, NPJT_FUNCTION)),
082                    sql, Executors.newCachedThreadPool(), mapper, 1024, 60);
083        }

119        /**
120         * Most detailed constructor
121         *
122         * @param sources data sources (wrapped into JDBC templates) accessor
123         * @param sql query to execute using JdbcOperations
124         * @param mapperFactory will be used to get data from result sets
125         * @param exceptionHandler workers exception handler
126         * @param executor executor service to run parallel queries into
127         * @param bufferSize size of {@link ArrayBlockingQueue} data buffer
128         * @param maxDataPollWaitSeconds max time to wait for next piece of data from workers
129         */
130        public ParallelQueriesIterator(DataSourceAccessor<?, ?> sources,
131                                       String sql, ExecutorService executor, RowMapperFactory<T, ?> mapperFactory,
132                                       ParallelQueriesExceptionHandler exceptionHandler, int bufferSize, int maxDataPollWaitSeconds) {
133            checkNotNull(sources, "Provided data source accessor is null");
134            checkArgument(sources.size() > 0, "No data sources provided");
135            checkArgument(hasText(sql), "Provided sql query is blank");
136            checkNotNull(executor, "Provided executor is null");
137            checkNotNull(mapperFactory, "Provided row mapper factory is null");
138            checkNotNull(exceptionHandler, "Provided exception handler is null");
139            checkArgument(bufferSize > 0, "Buffer size mat be positive, but was: '%s'", bufferSize);
140            this.sources = sources;
141            this.sql = sql;
142            this.mapperFactory = mapperFactory;
143            this.exceptionHandler = exceptionHandler;
144            this.executor = executor;
145            this.dataQueue = new ArrayBlockingQueue<Object>(bufferSize);
146            this.maxDataPollWaitSeconds = maxDataPollWaitSeconds;
147        }
148    
149        /**
150         * Starts parallel query execution in data sources. May be called multiple times to reuse iterator instance,
151         * Must be called only on exhausted iterators (which have all worker threads finished).
152         * May not be called on cancelled iterators.
153         *
154         * @param params query params
155         * @return iterator itself
156         */
157        public ParallelQueriesIterator<T> start(Collection<? extends SqlParameterSource> params) {
158            checkNotNull(params, "Provided parameters collection is null");
159            checkArgument(params.size() > 0, "Provided collection is empty");
160            checkState(!cancelled.get(), "This iterator is cancelled and cannot be restarted");
161            this.dataQueue.clear();
162            this.sourcesRemained.set(params.size());
163            this.futures = ImmutableList.copyOf(Collections2.transform(params, new SubmitFun()));
164            this.started.set(true);
165            return this;
166        }
167    
168        /**
169         * Cancels queries processing in all sources, by setting cancelled state, then cancelling all active
170         * prepared statements (for workers blocked on query completion) and then interrupting
171         * all threads (for workers blocked on result queue).
172         * May be called from another thread. Subsequent calls do nothing.
173         *
174         * @return count of threads that were actually interrupted in processing
175         */
176        public int cancel() {
177            if(!started.get()) return 0;
178            // set cancelled state
179            boolean wasCancelled = cancelled.getAndSet(true);
180            if(wasCancelled) return 0;
181            exceptionHolder.set(new ParallelQueriesException("cancelled"));
182            // cancel statements
183            for(Statement st : activeStatements.values()) {
184                cancelStatement(st);
185            }
186            int res = 0;
187            // cancel threads
188            for(Future<?> fu : futures) {
189                if(fu.cancel(true)) res += 1;
190            }
191            return res;
192        }
193    
194        /**
195         * Register listener for "query success" and "query error" events
196         *
197         * @param listener data source query events will be reported to this listener
198         * @return iterator itself
199         */
200        public ParallelQueriesIterator<T> addListener(ParallelQueriesListener listener) {
201            this.listeners.add(listener);
202            return this;
203        }

227        /**
228         * {@inheritDoc}
229         */
230        @Override
231        @SuppressWarnings("unchecked")
232        protected T computeNext() {
233            checkState(started.get(), "Iterator wasn't started, call 'start' method first");
234            Object ob;
235            while(endOfDataObject == (ob = takeData())) {
236                if(0 == sourcesRemained.decrementAndGet()) return endOfData();
237            }
238            return (T) ob;
239        }
240    
241        private void putData(Object data) {
242            try {
243                dataQueue.put(data);
244            } catch(InterruptedException e) {
245                Thread.currentThread().interrupt();
246                throw new RuntimeException(e);
247            }
248        }
249    
250        private Object takeData() {
251            try {
252                for(int i=0; i< maxDataPollWaitSeconds; i++) {
253                    checkException();
254                    Object res = dataQueue.poll(1, TimeUnit.SECONDS);
255                    if(null != res) return res;
256                }
257                throw new ParallelQueriesException(
258                        "Data wait timeout exceeded: '" + maxDataPollWaitSeconds + " seconds', alive workers count: '" + sourcesRemained.get() + "'");
259            } catch(InterruptedException e) {
260                Thread.currentThread().interrupt();
261                throw new ParallelQueriesException(e);
262            }
263        }
264    
265        private void checkException() {
266            ParallelQueriesException workerException = exceptionHolder.get();
267            if (null != workerException) {
268                try {
269                    exceptionHandler.handle(workerException);
270                } catch (RuntimeException e) {
271                    cancel();
272                    throw e;
273                }
274            }
275        }
276    
277        private void cancelStatement(Statement stmt) {
278            try {
279                stmt.cancel();
280            } catch(Exception e) { // driver may throw different exceptions
281                logger.warn("Exception thrown on statement cancelling: '" + stmt + "'", e);
282            }
283        }

285        private class Worker implements Runnable {
286            private final NamedParameterJdbcOperations npjo;
287            private final SqlParameterSource params;
288    
289            private Worker(NamedParameterJdbcOperations npjo, SqlParameterSource params) {
290                this.npjo = npjo;
291                this.params = params;
292            }
293    
294            @Override
295            @SuppressWarnings("unchecked") // workaround to
296            // keep RowMapperFactory fully typed without additional
297            // generic arguments to iterator itself
298            public void run() {
299                String registryKey = Thread.currentThread().getName();
300                try {
301                    RowMapperFactory ungeneric = mapperFactory;
302                    RowMapper<T> mapper = ungeneric.produce(params);
303                    Extractor extractor = new Extractor(mapper);
304                    // sql parameters processing is the same as in NamedParameterJdbcTemplate
305                    PreparedStatementCreator psc = new CancellableStatementCreator(registryKey, activeStatements, sql, params);
306                    npjo.getJdbcOperations().query(psc, extractor);
307                    for(ParallelQueriesListener li : listeners) li.success(npjo, sql, params);
308                } catch (Throwable e) { // we do not believe to JDBC drivers' error reporting
309                    ParallelQueriesException pqe = e instanceof ParallelQueriesException ? (ParallelQueriesException) e : new ParallelQueriesException(e);
310                    exceptionHolder.set(pqe);
311                    for(ParallelQueriesListener li : listeners) li.error(npjo, sql, params, e);
312                } finally {
313                    activeStatements.remove(registryKey);
314                }
315            }
316        }
317    
318        private class Extractor implements ResultSetExtractor<Void> {
319            private final RowMapper<T> mapper;
320    
321            private Extractor(RowMapper<T> mapper) {
322                this.mapper = mapper;
323            }
324    
325            @Override
326            public Void extractData(ResultSet rs) throws SQLException, DataAccessException {
327                try {
328                    int rowNum = 0;
329                    while(rs.next()) {
330                        if(cancelled.get()) return null;
331                        Object obj = mapper.mapRow(rs, rowNum++);
332                        putData(obj);
333                    }
334                    if(cancelled.get()) return null;
335                    putData(endOfDataObject);
336                    return null;
337                } catch(Throwable e) { // we do not believe to JDBC drivers' error reporting
338                    if(e instanceof SQLException) throw (SQLException) e;
339                    throw new SQLException(e);
340                }
341            }
342        }
343    
344        private class SubmitFun implements Function<SqlParameterSource, Future<?>> {
345            @Override
346            @SuppressWarnings("unchecked") // workaround to
347            // keep accessors fully typed without additional
348            // generic arguments to iterator itself
349            public Future<?> apply(SqlParameterSource params) {
350                DataSourceAccessor ungeneric = sources;
351                NamedParameterJdbcOperations npjo = ungeneric.get(params);
352                Worker worker = new Worker(npjo, params);
353                return executor.submit(worker);
354            }
355        }
