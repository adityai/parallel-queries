001    package com.alexkasko.springjdbc.parallel;
002    
003    import org.springframework.jdbc.core.RowMapper;
004    import org.springframework.jdbc.core.namedparam.SqlParameterSource;
005    
006    /**
007     * Spring's row mappers haven't access to query params.
008     * This class may be used if such access is needed on result set mapping.
009     *
010     * @author alexkasko
011     * Date: 8/18/12
012     * @see ParallelQueriesIterator
013     */
014    public interface RowMapperFactory<T, P extends SqlParameterSource> {
015        /**
016         * Must produce row mapper. Will be called by {@link ParallelQueriesIterator} before every SQL query
017         *
018         * @param params query params
019         * @return row mapper for query results
020         */
021        RowMapper<T> produce(P params);
022    }
