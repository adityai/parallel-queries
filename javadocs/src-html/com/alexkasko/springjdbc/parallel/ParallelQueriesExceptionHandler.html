001    package com.alexkasko.springjdbc.parallel;
002    
003    /**
004     * Exception handler for exceptions propagated from worker threads to caller thread.
005     *
006     * @author alexkasko
007     * Date: 1/29/13
008     */
009    public interface ParallelQueriesExceptionHandler {
010        /**
011         * If implementation rethrow exception (of throw another exception) workers execution
012          * will be cancelled and exception will be propagated to {@link java.util.Iterator#next()}
013          * or {@link java.util.Iterator#hasNext()} call.
014          * Will be called from the iterator caller thread on worker exception.
015         *
016         * @param e worker exception wrapped into {@link ParallelQueriesException}
017         * @throws RuntimeException rethrow or throw new exception to cancel workers execution
018         */
019        void handle(ParallelQueriesException e) throws RuntimeException;
020    }
