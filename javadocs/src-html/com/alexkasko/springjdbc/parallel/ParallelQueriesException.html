001    package com.alexkasko.springjdbc.parallel;
002    
003    import org.springframework.dao.DataAccessException;
004    
005    import static java.lang.Thread.currentThread;
006    
007    /**
008     * Exception for errors in async query workers
009     *
010     * @author alexkasko
011     * Date: 6/12/12
012     * @see ParallelQueriesIterator
013     */
014    public class ParallelQueriesException extends DataAccessException {
015    
016        ParallelQueriesException(String msg) {
017            super("Thread: '" + currentThread().getName() + "', message: '" + msg + "'");
018        }
019    
020        ParallelQueriesException(Throwable cause) {
021            super("Thread: '" + currentThread().getName() + "', message: '" + cause.getMessage() + "'", cause);
022        }
023    }
