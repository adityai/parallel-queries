001    package com.alexkasko.springjdbc.parallel.accessor;
002    
003    import org.springframework.jdbc.core.namedparam.NamedParameterJdbcTemplate;
004    
005    import java.util.Collection;
006    
007    import static java.util.Arrays.asList;
008    
009    /**
010     * {@link RoundRobinAccessor} subclass for {@code NamedParameterJdbcTemplate}.
011     *
012     * @author alexkasko
013     * Date: 1/21/13
014     * @since 1.1
015     * @see DataSourceAccessor
016     */
017    public class RoundRobinNpjtAccessor extends RoundRobinAccessor<NamedParameterJdbcTemplate> {
018    
019        /**
020         * Vararg constructor
021         *
022         * @param jts data sources
023         */
024        public RoundRobinNpjtAccessor(NamedParameterJdbcTemplate... jts) {
025            this(asList(jts));
026        }
027    
028        /**
029         * Main constructor
030         *
031         * @param target data sources
032         */
033        public RoundRobinNpjtAccessor(Collection<NamedParameterJdbcTemplate> target) {
034            super(target);
035        }
036    }
